        contractDetails:{
            available_from:"End of year 1",
            commissioning: 1383091200000,

                       action) => {
    switch (action.type){
        case AppConstants.LOADING_CONTRACT_OVERVIEW_DATA:
            return Object.assign({}, state, {loadingOverViewData: true})
        case AppConstants.LOADED_CONTRACT_OVERVIEW_DATA:
            return Object.assign({}, state, {loadingOverViewData: false, overviewData: List(action.payload.overviewData)})
        case AppConstants.LOADING_CONTRACT_FIELDS_DATA:
            return Object.assign({}, state, {loadingSelectedContractData: true})
        case AppConstants.LOADED_CONTRACT_FIELDS_DATA:
            return Object.assign({}, state, {loadingSelectedContractData: false, selectedContractData: action.payload.selectedContractData})
        case AppConstants.SET_CONTRACT:
            return Object.assign({}, state, {selectedContractId: action.payload.selectedContractId});
        case AppConstants.SET_CURRENTLY_SHOWING:
            return Object.assign({}, state, {currentlyShowing: action.payload.show})
        case AppConstants.SET_WIND_PARK:
            return Object.assign({}, state, {selectedWindPark: action.payload.windPark})
        case AppConstants.SET_CONTRACT_DETAILS:
            return Object.assign({},state,{contractDetails:action.payload.details});
        case AppConstants.SET_CONTRACT_YEAR_DETAILS:
            return Object.assign({},state,{contractYearDetails:action.payload.details})
        case AppConstants.SET_TURBINE_CONTRACT_DETAILS:
            return Object.assign({},state, {turbineContractDetails:action.payload.details});
        default:
            return state
 
 
 
 
 
 
    }
}
