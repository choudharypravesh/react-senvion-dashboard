import AppDispatcher from '../../dispatcher/AppDispatcher'
import AppConstants from '../../constants/AppConstants';
import {ReduceStore} from 'flux/utils';
import Immutable, {Map} from 'immutable';
import ReactCookie from 'react-cookie';
import moment from 'moment';
//import {AppAPI} from '../Utils/utils';
 
export default class DashboardStore extends ReduceStore {
 
  getInitialState() {
    return Map({
        agX: ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun',
            'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec'],
        agY1: [20, 14, 25, 16, 18, 22, 19, 15, 12, 16, 14, 17],
        agY2: [19, 14, 22, 14, 16, 19, 15, 14, 10, 12, 12, 16],
        ttY1: ["Achairn", "Achany", "A'Chruach", "Achtrup", "Ahlen", "Aino Cho", "Airaines", "Beseritz"],
        ttY2: ["Bestepe", "Bickenbach", "Bierbergen", "Bilsthorpe", "Birkenzell", "Bisaccia", "Blantyre Muir", "Blindow"],
        wtY1: ["90987-", "90765-", "96475-", "91637-", "91745-", "91117-", "98877-", "95564-"],
        user: ReactCookie.load('user'),
        bestFarmsArray: [],
        worstFarmsArray: [],
        bestTurbinesArray: [],
        worstTurbinesArray: [],
        totalAlerts: 0,
        openAlerts: 0,
        alertsByTimeLoader: true,
        alertsByTimeNoData: false,
        alertsByTimeX: [],
        alertsByTimeY: [],
        availabilityLoader: true,
        availabilityNoData: false,
        availabilityX: [],
        availabilityY1: [],
        availabilityY2: []
    });
  }
 
  reduce(state, action) {
    switch (action.type) {
      case AppConstants.DASHBOARD_LOADED:{
        switch(action.category){
            case AppConstants.DASHBOARD_ALERTSTIME_DATA:
                return state.set('alertsByTimeLoader', action.loaderStatus)
                        .set('alertsByTimeNoData', action.noData)
                        .set('alertsByTimeX', action.x)
                        .set('alertsByTimeY', action.y)
            case AppConstants.DASHBOARD_AVAILABILITY_DATA:
                return state.set('availabilityLoader', action.loaderStatus)
                            .set('availabilityNoData', action.noData)
                            .set('availabilityX', action.x)
                            .set('availabilityY1', action.y1)
                            .set('availabilityY2', action.y2)
            case AppConstants.BEST_FARMS_DATA:
                return state.set('bestFarmsArray', action.bestFarmsArray)
 
            case AppConstants.BEST_TURBINES_DATA:
                return state.set('bestTurbinesArray', action.bestTurbinesArray)
            case AppConstants.WORST_FARMS_DATA:
                return state.set('worstFarmsArray', action.worstFarmsArray)
            case AppConstants.WORST_TURBINES_DATA:
                return state.set('worstTurbinesArray', action.worstTurbinesArray)
            case AppConstants.ALERT_COUNT_DATA:
                return state.set('openAlerts', action.openAlerts).set('totalAlerts', action.totalAlerts)
        }
         break;
      }
      case AppConstants.DASHBOARD_LOADING:{
        switch (action.category){
            case AppConstants.DASHBOARD_ALERTSTIME_DATA:
                return state.set('alertsByTimeLoader', action.loaderStatus).set('alertsByTimeNoData', action.noData);
            case AppConstants.DASHBOARD_AVAILABILITY_DATA:
                return state.set('availabilityLoader', action.loaderStatus).set('availabilityNoData', action.noData);
        }
        break;
      }
//        case AppConstants.NODATA:
//        if(action.category === 'all'){
//            return state.set('allNoData', action.noData).set('allDataLoader', action.loaderStatus);
//        }else{
//            return state.set('hubNoData', action.noData).set('hubDataLoader', action.loaderStatus);
//        }
      default:
        return getInitialState();
    }
  }
}
