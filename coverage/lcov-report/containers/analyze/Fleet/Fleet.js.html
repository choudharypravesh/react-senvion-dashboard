// App.js
import React from 'react';
/*import 'react-dates/lib/css/_datepicker.css';*/
import 'react-virtualized-select/styles.css';
import classnames from 'classnames';
import AvailabilityTime from './AvailabilityTime';
import AvailabilityGen from './AvailabilityGen';
import {AppActions} from '../../../actions/AppActions';
import { Dispatcher } from 'flux';
import FleetStore from './FleetStore';
 
 
class Fleet extends React.Component{
    constructor(props){
        super(props)
        this.dispatcher = new Dispatcher();
        this.store = new FleetStore(this.dispatcher);
        this.actions = new AppActions(this.dispatcher);
        this.state = {data: this.store.getState()}
    }
    componentDidMount() {
        this.listener = this.store.addListener(this.onStateChanged.bind(this));
    }
    componentWillUnmount() {
        this.listener.remove();
    }
    onStateChanged() {
        this.setState({ data: this.store.getState() });
    }
    getTabContent(){
        switch(this.state.data.get('selectedTab')) {
            case 1 : return(
                <AvailabilityTime data={this.state.data} actions={this.actions}/>
            );
            case 2 : return(
                <AvailabilityGen data={this.state.data} actions={this.actions}/>
            );
        }
    }
    render(){
        return (
            <div className="portfolio-container" id="fleet-graphs">
                <div className="fleet-tabs margin-top level-1-tabs">
                    <div className="btn-group row" role="group">
                        <button className={classnames("btn btn-default btn1 col-xs-4", {"active": this.state.data.get('selectedTab') === 1})} onClick={() => {this.actions.changeFleetTab(1)}}>Availability(Time)</button>
                        <button className={classnames("btn btn-default btn2 col-xs-4", {"active": this.state.data.get('selectedTab') === 2})} onClick={() => {this.actions.changeFleetTab(2)}}>Availability(Gen.)</button>
                        <button className="btn btn-default col-xs-4">Performance KPIs</button>
                    </div>
                </div>
 
                {this.getTabContent()}
 
            </div>
        );
    }
}
 
 
 
export default Fleet;
