import { combineReducers } from 'redux'
import moment from 'moment';
import {Map} from 'immutable'
import AppConstants from '../../../constants/AppConstants'
const farmData = (state = {
                    selectedTab: 1,
                    selectedCountry: "",
                    selectedFarm:"",
                    initialDate: moment().subtract(7, 'days').format('YYYY-MM-DD'),
                    finalDate: moment().format('YYYY-MM-DD'),
                    countriesList: [],
                    farmsList: [],
                    power: 0,
                    windSpeed: 0,
                    availTime: 0,
                    availGen: 0,
                    lpf: 0,
                    numberOfAlerts: 0
                }, action) => {
    switch(action.type){
        case AppConstants.SET_DATES:
            return Object.assign({}, state, {initialDate: action.payload.initialDate, finalDate: action.payload.finalDate})
        case AppConstants.SELECT_TAB:
            return Object.assign({},state, {selectedTab: action.payload.id})
        case AppConstants.SELECT_COUNTRY:
            return Object.assign({},state, {selectedCountry: action.payload.countryId});
        case AppConstants.RECEIVED_COUNTRIES_LIST:
            return Object.assign({},state, {countriesList: action.payload.countriesList, selectedCountry: action.payload.countriesList[0].value});
        case AppConstants.SELECT_FARM:
            return Object.assign({}, state, {selectedFarm: action.payload.farm})
        case AppConstants.RECEIVED_FARMS_LIST:
            return Object.assign({}, state, {farmsList: action.payload.farmsList, selectedFarm: action.payload.farmsList[0].value})
        case AppConstants.RECEIVED_POWER_WIND:
            return Object.assign({}, state, {power: action.payload.power, windSpeed: action.payload.wind});
        case AppConstants.RECEIVED_AVAIL_COUNT:
            return Object.assign({}, state, {availGen: action.payload.availGen, availTime: action.payload.availTime, lpf: action.payload.lpf})
        case AppConstants.RECEIVED_ALERTS_COUNT:
            return Object.assign({}, state, {numberOfAlerts: action.payload.numberOfAlerts})
        case AppConstants.LOADING_AVAILABILITY:
            return Object.assign({}, state, {availLoaderStatus: action.payload.loaderStatus});
        case AppConstants.LOADED_AVAILABILITY:
            return Object.assign({}, state, {availLoaderStatus: action.payload.loaderStatus, availabilityX: action.payload.availabilityX, availabilityY: action.payload.availabilityY});
        case AppConstants.LOADED_PBA:
            return Object.assign({}, state, {availLoaderStatus: action.payload.loaderStatus, farmPbaX: action.payload.farmPbaX, farmPbaY: action.payload.farmPbaY})
        case AppConstants.LOADING_WIND_DATA:
            return Object.assign({}, state, {windDataLoader: action.payload.loaderStatus})
        case AppConstants.LOADED_WIND_DATA:
            return Object.assign({}, state, {windDataLoader: action.payload.loaderStatus, wsX: action.payload.wsX, wsY: action.payload.wsY, wdX: action.payload.wdX, wdY: action.payload.wdY})
        case AppConstants.SET_VARIABLE:
            if(action.payload.chart == 1){
                return Object.assign({}, state, {selectedVariable1: action.payload.variable})
            }else{
                return Object.assign({}, state, {selectedVariable2: action.payload.variable})
            }
        case AppConstants.RECEIVED_VARIABLES:
            return Object.assign({}, state, {variablesList: action.payload.variablesList, selectedVariable1: action.payload.variablesList[0].value, selectedVariable2: action.payload.variablesList[1].value})
        case AppConstants.LOADING_TURBINE_HEAT_MAP_DATA:
            if(action.payload.chart == 1){
                return Object.assign({}, state, {heatMapLoader: action.payload.loaderStatus})
            }else{
                return Object.assign({}, state, {heatMap2Loader: action.payload.loaderStatus})
            }
        case AppConstants.RECEIVED_TURBINE_HEAT_MAP_DATA:
            if(action.payload.chart == 1){
                return Object.assign({}, state, {heatMapLoader: action.payload.loaderStatus, heatMapX: action.payload.x, heatMapY: action.payload.y, heatMapZ: action.payload.z})
            }else{
                return Object.assign({}, state, {heatMap2Loader: action.payload.loaderStatus, heatMap2X: action.payload.x, heatMap2Y: action.payload.y, heatMap2Z: action.payload.z})
            }
        default:
            return state;
    }
}
 
export default farmData;
